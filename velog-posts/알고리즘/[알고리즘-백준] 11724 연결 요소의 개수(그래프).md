### 문제
dfs 또는 bfs 구현 문제이다.

입력으로 주어지는 간선의 양 끝 점을 연결하면 몇 개의 연결 요소가 나오는지 계산하는 문제이다.
참고: https://dduniverse.tistory.com/entry/%EB%B0%B1%EC%A4%80-11724-%EC%97%B0%EA%B2%B0-%EC%9A%94%EC%86%8C%EC%9D%98-%EA%B0%9C%EC%88%98-%ED%8C%8C%EC%9D%B4%EC%8D%AC-python

### 풀이(DFS)
```python
n, m = map(int, input().split())

def dfs(num): 
    if visit[num] == -1: # 방문 했다면 return
        return
    visit[num] = -1 # 방문 표시
    for i in range(len(graph[num])): # 깊이 탐색
        dfs(graph[num][i])

graph = [[] for x in range(n+1)]
visit = [0]*(n+1) # 방문 표시할 리스트 생성
for i in range(m): # 그래프 양방향 리스트로 만들기
    u, v = map(int, input().split())
    graph[u].append(v)
    graph[v].append(u)

cnt = 0
for i in range(1, n+1): 
    if visit[i] == 0: # 방문하지 않은 점이라면
        dfs(i) # 방문
        cnt+=1 # 카운트 +1
print(cnt)
```

### 풀이(BFS)
```python
from collections import deque

n, m = map(int, input().split())

def bfs(num):
    q = deque([]) # 방문 리스트
    q.append(num) 
    while q: # q가 비어있지 않을 때
        num = q.popleft() # 점 하나 꺼내서
        for i in graph[num]: # 방문 했는지 확인
            if visit[i] == -1: # 이미 방문한 점이라면 q에 추가하지않기
                continue 
            else:
                q.append(i) # q에 방문할 점 추가
                visit[i] = -1 # 방문 표시

graph = [[] for x in range(n+1)]
visit = [0]*(n+1)
for i in range(m): # 양방향 리스트 만들기
    u, v = map(int, input().split())
    graph[u].append(v)
    graph[v].append(u)

cnt = 0
for i in range(1, n+1):
    if visit[i] == 0: # i가 방문하지 않은 점이라면 bfs 실행
        bfs(i)
        cnt+=1 # 카운트 추가
print(cnt)
```
